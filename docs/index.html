    --primary: #121111;

    --accent: #BDBBBB;
    --neutral: #555454;
    --neutral-inverse: #f1f1f1;
    --danger: #D9422B;
    --success: #689342;
    --warning: #E66700;
    --sans-serif: 'Segoe UI', Tahoma, Geneva, Verdana, sans-serif;
    --serif: Georgia, 'Times New Roman', Times, serif;
    --code: 'Courier New', Courier, monospace;
}
